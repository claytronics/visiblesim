60  ss << "Message from VM : END_SIMULATION";
61  break;
62  case VM_MESSAGE_TYPE_CREATE_LINK:
63  ss << "Message from VM : CREATE_LINK ";
64  ss << "from " << param1 << " to " << param2;
65  break;
66  case VM_MESSAGE_TYPE_COMPUTATION_LOCK:
67  ss << "Message from VM : COMPUTATION_LOCK ";
68  ss << "to block " << param1 << " for " << param2 << " microseconds";
69  break;
70  case VM_MESSAGE_TYPE_COMPUTATION_UNLOCK:
71  ss << "Message to VM : COMPUTATION_UNLOCK ";
72  ss << "from block " << param1 << " at t=" << param2;
73  break;
74  case VM_MESSAGE_TYPE_SEND_MESSAGE:
75  ss << "Message from VM : SEND_MESSAGE ";
76  ss << param3 << " bytes from block " << param1 << " to block " << param2;
77  break;
78  case VM_MESSAGE_TYPE_RECEIVE_MESSAGE:
79  ss << "Message to VM : RECEIVE_MESSAGE ";
80  ss << "block " << param1 << " received " << param3 << " bytes at t=" << param2;
81  break;
82  case VM_MESSAGE_TYPE_WAIT_FOR_MESSAGE:
83  ss << "Message from VM : WAIT_FOR_MESSAGE ";
84  ss << "block " << param1;
85  break;
86  default:
87  ss << " *** UNKNOWN MESSAGE TYPE ***";
88  break;
